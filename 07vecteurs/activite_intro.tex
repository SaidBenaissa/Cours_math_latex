\begin{exercice}
Une télécabine se déplace sur un cable, de $A$ à $B$.
\begin{multicols}{2}
  \begin{center}
    \begin{tikzpicture}[very thick,scale=.6]
        \draw[dotted, gray] (0,0) grid (12, 7);
        \draw (2, 0)  node[below left]{$C$}
         -- ++(2, 0)  node[below right]{$D$}
         -- ++(0, 3)  node[above right]{$E$}
         -- ++(-2, 0) node[above left]{$F$}
         -- cycle;
         \draw (3,3) -- (3,4);
         \draw (0,3) -- (12, 7);

         \draw (3,4) node{$\bullet$} node[above]{$A$};
         \draw (9,6) node{$\bullet$} node[above]{$B$};
    \end{tikzpicture}
  \end{center}

  \begin{enumerate}
    \item Dessiner la télécabine lorsqu'elle sera arrivée au terminus $B$.
    \item Comment décrire ce déplacement ?
    \item Représenter ce déplacement sur le graphique.
  \end{enumerate}
  
\end{multicols}
\end{exercice}
